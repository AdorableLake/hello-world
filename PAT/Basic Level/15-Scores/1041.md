## 1041. 考试座位号

### 《算法笔记》
```C
#include<stdio.h>
const int MAXN=1010;
struct Student
{
    long long int id;
    int examSeat;
}
testSeat[MAXN];

int main()
{
    int N,M,seat,examSeat;
    long long int id;
    scanf("%d\n",&N);
    
    for(int i=0;i<N;i++)
    {
        scanf("%lld %d %d\n",&id,&seat,&examSeat);
        testseat[seat].id=id;
        testseat[seat].examseat=examseat;
    }
    scanf("%d\n",&M);
    for(int i=0;i<M;i++)
    {
        scanf("%d",&seat);
        printf("%lld %d\n",testseat[seat].id,testseat[seat].examseat);
        if(i<M-1) printf(" ");
    }
    return 0;
}
```
