```C
#include<stdio.h>
#include<math.h>
int main()
{
  int A,Da,B,Db;
  scanf("%d %d %d %d",&A,&Da,&B,&Db);
  /*
  printf("A=%d Da=%d B=%d Db=%d\n",A,Da,B,Db);
  */
  //利用for循环统计A的位数
  int na=0,nb=0;
  for(;A>9;)
  {
    int temp;
    temp=A%10;
    if(temp==Da)//如果所的位数等于Da
    {
      na++;
    }
    A=A/10;
  }
  //利用for循环统计B的位数
  for(;B>9;)
  {
    int temp=B;
    temp=B%10;
    if(temp==Db)
    {
      nb++;
    }
    B=B/10;
  }
  int Pa=0,Pb=0;
  for(int i=1;i<=na;i++)
  {
    Pa=Pa+Da*pow(10,i-1);
  }
    
  for(int i=1;i<=nb;i++)
  {
    Pb=Pb+Db*pow(10,i-1);
  }
  printf("%d\n",Pa+Pb);
    
  return 0;
}
```
