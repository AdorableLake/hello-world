#include<stdio.h>
int main()
{
  int i,j,N,M,temp;
  scanf("%d%d",&N,&M);
  M %= N;
  int A[N];
  for(i=0;i<N;i++)
  {
    scanf("%d",&A[i]);
  }
  for(i=0;i<M;i++)
  {
    temp = A[N-1];
    for(j=N-1;j>0;j--)
    {
      A[j] = A[j-1];
    }
    A[0] = temp;
  }
  for(i=0;i<N-1;i++)
  {
    printf("%d ",A[i]);
  }
  printf("%d",A[N-1]);
  return 0;
}
